# Vue下载后台返回的文件流

Veu项目下载文件，后端返回文件流，axios发送下载请求下载文件

这里用到了Blob对象，这里是从服务器接收到的文件流（content-type:application/octet-stream）创建blob对象并使用该blob 创建一个指向类型数组的URL，将该url作为a标签的链接目标，然后去触发a标签的点击事件从而文件下载

downloadWord(id).then((res) => {
        let blob = new Blob([res.data], {type: "application/vnd.openxmlformats-officedocument.wordprocessingml.document"})
        let objectUrl = URL.createObjectURL(blob) // 创建URL
        location.href = objectUrl;
        URL.revokeObjectURL(objectUrl); // 释放内存
  });

type这里application/后的参数是根据下载的文件类型来修改，本例是下载docx格式的文档，具体查看 MIME 类型列表 (opens new window)

也可以写个方法调用：

/**
data: 后台返回的res.data
fileName：下载文件的文件名
*/
download (data,fileName) {
    if (!data) {
        return this.$message('无相关文件下载地址')
    }
    let url = window.URL.createObjectURL(new Blob([data], {type: "application/msword"})) //创建下载链接
    let link = document.createElement('a') //创建a标签
    link.style.display = 'none'  //将a标签隐藏
    link.href = url  //给a标签添加下载链接
    link.setAttribute('download', fileName) // 此处注意，要给a标签添加一个download属性，属性值就是文件名称 否则下载出来的文件是没有属性的，空白白
    document.body.appendChild(link)  
    link.click()  //执行a标签
    URL.revokeObjectURL(url)
}

// 下载按钮点击事件
exportClick () {
      export().then(res => {
        const content = res
        const blob = new Blob([content])
        const fileName = '导出信息.xlsx'
        if ('download' in document.createElement('a')) { // 非IE下载
          const elink = document.createElement('a')
          elink.download = fileName
          elink.style.display = 'none'
          elink.href = URL.createObjectURL(blob)
          document.body.appendChild(elink)
          elink.click()
          URL.revokeObjectURL(elink.href) // 释放URL 对象
          document.body.removeChild(elink)
        } else { // IE10+下载
          navigator.msSaveBlob(blob, fileName)
        }
      })
    }
