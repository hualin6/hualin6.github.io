Vue仪表盘的使用

MarkLin 2021-04-01 Vue

Vue使用仪表盘

本案例使用Viser 插件实现, 与antV的G2 基本一致, 因为Viser 一个基于 G2 实现的，为数据可视化工程师量身定制的工具。使用前安装该插件, 具体使用参考官网文档 (opens new window)及API

npm install viser-vue
1

一般是以一个单独的.vue文件,需要的页面引入来使用

传入数值score即可


<template>
  <div class="safa-status">
    <v-chart :forceFit="false" :height="height" :width="330" :padding='padding' :data="data" :scale="scale"  class="content">
      <v-coord type="polar" :startAngle="-202.5" :endAngle="22.5" :radius="0.6" ></v-coord>
      <v-axis
        dataKey="value"
        :zIndex="2"
        :line="null"
        :label="axisLabel"
        :tickLine="null"
        :grid="null"
      ></v-axis>
      <v-axis dataKey="1" :show="false"></v-axis>
      <v-series
        gemo="point"
        position="value*1"
        shape="pointer"
        color="#1890FF"
        :active="false"
      ></v-series>
      <v-guide
        type="line"
        :start="lineGuide1Start"
        :end="lineGuide1End"
        :lineStyle="lineGuide1LineStyle"
      ></v-guide>
      <v-guide
        type="line"
        :start="lineGuide2Start"
        :end="lineGuide2End"
        :lineStyle="lineGuide2LineStyle"
      ></v-guide>
      <v-guide
        type="line"
        :start="lineGuide3Start"
        :end="lineGuide3End"
        :lineStyle="lineGuide3LineStyle"
      ></v-guide>
      <v-guide
        type="line"
        :start="lineGuide4Start"
        :end="lineGuide4End"
        :lineStyle="lineGuide4LineStyle"
      ></v-guide>
      <v-guide
        type="arc"
        :zIndex="0"
        :top="false"
        :start="arcGuide1Start"
        :end="arcGuide1End"
        :vStyle="arcGuide1Style"
      ></v-guide>
      <v-guide
        type="arc"
        :zIndex="1"
        :start="arcGuide2Start"
        :end="getArcGuide2End"
        :vStyle="arcGuide2Style"
      ></v-guide>
      <v-guide
        type="html"
        :position="htmlGuidePosition"
        :html="getHtmlGuideHtml()"
      ></v-guide>
    </v-chart>
  </div>
</template>

<script>
import { registerShape } from 'viser-vue';
registerShape('point', 'pointer', {
  draw(cfg, container) {
    let point = cfg.points[0]; // 获取第一个标记点
    point = this.parsePoint(point);
    const center = this.parsePoint({ // 获取极坐标系下画布中心点
      x: 0,
      y: 0
    });
    // 绘制指针
    container.addShape('line', {
      attrs: {
        x1: center.x,
        y1: center.y,
        x2: point.x,
        y2: point.y,
        stroke: cfg.color,
        lineWidth: 2,
        lineCap: 'round'
      }
    });
    return container.addShape('circle', {
      attrs: {
        x: center.x,
        y: center.y,
        r: 4.75,
        stroke: cfg.color,
        lineWidth: 2,
        fill: '#fff'
      }
    });
  }
});
// 设置ABCDE标识位置
const scale = [{
  dataKey: 'value',
  min: 0,
  max: 9,
  ticks: [1, 2.8, 4.5, 6.2,8],
  nice: false
}];

const data = [
  { value: 9 }
];

export default {
  props:{
    score:{
      type:Number,
      default:95
    }
  },
  data() {
    return {
      padding: [ -100, 0, 0, 0],
      height: 420,
      data: data,
      scale: scale,
      // 坐标轴
      axisLabel: {
        offset: -20,
        formatter: (val) => {
          if (val === '1') {
            return 'E';
          } else if (val ==='2.8') {
            return 'D';
          } else if (val ==='4.5') {
            return 'C';
          } else if (val === '6.2') {
            return 'B';
          }
          return 'A';
        },
        textStyle: {
          fontSize: 16,
          textAlign: 'center'
        }
      },
      // 刻度线
      lineGuide1Start: [2, 0.905],
      lineGuide1End: [2.0035, 0.8],
      lineGuide1LineStyle: {
        stroke: '#19AFFA',
        lineDash: null,
        lineWidth: 2
      },

      lineGuide2Start: [3.66, 0.905],
      lineGuide2End: [3.66, 0.8],
      lineGuide2LineStyle: {
        stroke: '#19AFFA',
        lineDash: null,
        lineWidth: 2
      },

      lineGuide3Start: [5.32, 0.905],
      lineGuide3End: [5.32, 0.8],
      lineGuide3LineStyle: {
        stroke: '#19AFFA',
        lineDash: null,
        lineWidth: 2
      },
      lineGuide4Start: [7, 0.905],
      lineGuide4End: [7, 0.8],
      lineGuide4LineStyle: {
        stroke: '#19AFFA',
        lineDash: null,
        lineWidth: 2
      },
      // 灰色部分
      arcGuide1Start: [0, 0.945],
      arcGuide1End: [9, 0.945],
      arcGuide1Style: {
        stroke: '#CBCBCB',
        lineWidth: 10,
      },
      //  蓝色部分
      arcGuide2Start: [0, 0.945],
      // arcGuide2End: [data[0].value, 0.945],
      arcGuide2Style: {
        stroke: '#1890FF',
        lineWidth: 10,
      },
      // 文字位置
      htmlGuidePosition: ['50%', '75%'],
    };
  },
  created() {
    if(this.score || this.score == 0){
      let val = this.score
      if(val<50){
        this.data[0].value = 1
      }else if(val < 66){
        this.data[0].value = 2.8
      }else if(val < 80){
        this.data[0].value = 4.5
      }else if(val < 90){
        this.data[0].value = 6.2
      }else{
        this.data[0].value = (val-10)/10
      }
    }
  },
  watch: {
  },

  methods: {
    // 仪表盘文字
    getHtmlGuideHtml(){
      return `
        <div style="width: 330px;text-align: center;">
          <p style="font-size: 20px;color: #545454;margin: 0;">分数</p>
          <p style="font-size: 36px;color: #545454;margin: 0;">${this.score}</p>
        </div>
      `
    },
    getArcGuide2End(){
      return [this.data[0].value, 0.945]
    },
  },
};
</script>
