MarkLin 2021-04-13 Vue

百度地图简单使用

html

<div class="bm-div" :key="timer">
    <baidu-map :center="center" :zoom="zoom" @ready="handler" class="bm-view">
    	<bm-view style="flex: 1"/>
    </baidu-map>
</div>

created

// 获取经纬度
let data =this.$store.getters.viewDetails
// 如果有经纬度数据就使用已有数据展示，如果没有就指定地图中心来显示
if(data){
	this.locationModal.search =  data.name
	this.locationModal.coordinate.longitude = data.gpslog
	this.locationModal.coordinate.latitude = data.gpslan
}else{
	this.center= {lng: 113.5, lat: 25.3}
}

methods

// 百度地图初始化
      handler ({BMap, map}) {
        let that = this;
        that.map = map;
        setTimeout(() => {
          map.centerAndZoom(that.locationModal.search, 16)
        },800)
        // 添加地图控制
        map.addControl(new BMap.MapTypeControl({
          mapTypes:[
            BMAP_NORMAL_MAP,
            BMAP_HYBRID_MAP
          ]
        }));
        // 是否启用缩放功能
        map.enableScrollWheelZoom(true);
        // 添加标记
        if (that.locationModal.coordinate){
          setTimeout(()=>{
            that.addTag();
          },1200);
        }
        // 给地图添加点击事件
        map.addEventListener('click', function (e) {
          that.mapChange = true;
          that.locationModal.coordinate.latitude = e.point.lat;
          that.locationModal.coordinate.longitude = e.point.lng;
          that.addTag();
        });
        // 如果没有经纬度数据，并且搜索到的值有数据，则以该数据为显示中心，否则指定显示广东地图
        setTimeout(() => {
          let lat = that.locationModal.coordinate.latitude
          if(!lat){
            if(that.map.lh && that.map.lh != '中国'){
              map.centerAndZoom(that.map.lh, 7)
            }else{
              map.centerAndZoom('广东', 7)
            }
          }
        },2000)
      },
      // 添加标记
      addTag() {
        let that = this;
        let lng = that.locationModal.coordinate.longitude
        let lat = that.locationModal.coordinate.latitude
        that.center.lng = lng
        that.center.lat = lat
        that.map.removeOverlay(that.markerPoint);
        let point = new BMap.Point(lng, lat);
        let marker = new BMap.Marker(point);
        that.markerPoint = marker;
        that.map.addOverlay(marker);
      },
