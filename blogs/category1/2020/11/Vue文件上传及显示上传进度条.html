data(){
    return{
        showProcess:false;
        progress:0
    }
}

onProgress(file,fileList){
      if(file.status === 'ready'){
        this.progress = 0;
        this.showProcess =true;
        const interval =setInterval(() => {
          if(this.progress >= 99){
            clearInterval(interval)
            return
          }
          this.progress +=1
        },20)
      }
      if(file.status === 'success'){
        this.progress = 100;
        this.showProcess = false
      }
    },

# JS部分

由于实际项目中上传的文件较大,500M左右,切片上传的话前后端要写的代码就很多了,现在前后端懒得太折腾故使用put方式上传文件.

前端先传送用户id获取上传地址,然后将上传文件并弹窗显示上传进度,最后上传完成后调用上传回调接口获取文件是否上传成功.

//data
upId:'',	//上传文件的用户id
Upurl:'',	//上传地址
upVisible:false,	//弹窗控制
upVal:0,	//进度条数值

// methods
// 获取上传地址
Fupload(val){
    this.upId = val
    getUploadUrl({id:val}).then(res => {
        if(res.code == '200'){
            this.Upurl= res.data
        }else{
            return this.$message.error('获取上传地址失败')
        }
    })
},
    
upChange(file, fileList){
    //弹窗显示上传进度
   this.upVisible = true;
,
//自定义请求头,覆盖elementui默认请求    
uploadF(val) {
    //定义文件格式校验规则
    let FileExt = val.file.name.replace(/.+\./, "").toLowerCase();
    //设置上传文件格式docx,若可以上传多种则添加到数组中即可['docx','doxc']
    let flag = ['docx'].includes(FileExt)
    if(flag){
        let formData = new FormData()
        let id = this.upId
        formData.append('file', val.file)
        this.$axios({
            url:this.Upurl,
            method:'put',
            data:formData,
            onUploadProgress: (progressEvent) => {
                let progressPercent = Number((progressEvent.loaded / progressEvent.total * 100).toFixed(2))
                this.upVal = progressPercent
            }
        }).then(res => {
            if (res.status == '200') {
                // 文件上传状态回调,通过这个后台接口返回文件是否上传成功
                getUplodBack({uploadId: id}).then(res => {
                    if(res.code == '200'){
                        this.$message.success(res.msg)
                    }else{
                        this.$message.error('上传失败')
                    }
                })
            } else {
                this.$message.error('文件上传失败')
            }
            this.upVisible = false
        }).catch((error) => {
            this.upVisible = false
            this.$message.error(error)
            console.log(error)
        });
    }else{
        return this.$message.error('只能上传‘.docx’格式文件，请重新选择文件上传!')
    }
},

 //上传
 beforeAvatarUpload(file) {
     if (file.size > 200 * 1024 * 1024) {
     	this.$message.error('文件不能超过200M');
     	return false
     }
 },
 //自定义请求头
 uploadImages(val) {
 	let formData = new FormData()
 	//key：file与后台对应
 	formData.append('file', val.file)
 	http.post(
 	'/file/upload',
 	formData,
 	{
 	headers: {'Content-Type': 'multipart/form-data'}
 	}
 	).then(res => {
         if (res.status==0) {
             let obj = {
             appendixName: res.data.filePath,
             appendixPath: res.data.message,
    	 }
 			this.ruleForm.announceAppendixList.push(obj);
 		 } else {
 			this.$message.error(res.message)
 		 }
 	}).catch((error) => {
     console.log(error)
     });
 },

//下载
handleList(file){
	let url='/file/downLoadToLocal';
    http.get(
    url,
    file.url ? {filePath: file.url} : {},
    {responseType: 'blob'}
    ).then(res => {
    let fileName = decodeURIComponent(res.headers['content-		disposition'].slice(res.headers['content-disposition'].indexOf('=') + 2));
    this.downloadFile(res.data, fileName);
    })
},

downloadFile(content, fileName) {
	(fileName && fileName.indexOf('.') !== -1) && (fileName = fileName.slice(0, 		fileName.indexOf('.')));
	const blob = new Blob([content], {
	type: content.type
	})
	if ('download' in document.createElement('a') && navigator.userAgent.indexOf('Edge') <= -1) { // 非IE 及edge下载
	const elink = document.createElement('a')
	fileName && (elink.download = fileName)
	elink.style.display = 'none'
	elink.href = URL.createObjectURL(blob)
	document.body.appendChild(elink)
	elink.click()
	URL.revokeObjectURL(elink.href) // 释放URL 对象
	document.body.removeChild(elink)
	} else { // IE10+下载
	fileName ? navigator.msSaveOrOpenBlob(blob, fileName) : navigator.msSaveOrOpenBlob(blob)
	}
},
